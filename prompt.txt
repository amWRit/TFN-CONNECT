FULL Prisma Models:

text
Person {
  id            String   @id @default(cuid())
  firstName     String
  lastName      String
  dob           DateTime
  phone1        String?
  phone2        String?
  email1        String?
  email2        String?
  linkedin      String?
  eduStatus     String   // 'enrolled'|'completed'|'not_enrolled'
  empStatus     String   // 'employed'|'seeking'|'unemployed'
  type          String   // 'alumni'|'LDC'|'admin'
  
  // Relations
  educations    Education[]
  experiences   Experience[]
  fellowships   Fellowship[]
  placements    Placement[]
  posts         Post[]
  ldcs          LDC[]     @relation("LDC_person")
}

Skill {
  id            String   @id @default(cuid())
  name          String   @unique // 'Leadership', 'Teaching', 'MS Suite'
  category      String   // 'technical'|'personal'|'interpersonal'
  experienceSkills ExperienceSkill[]
}

Experience {
  id            String   @id @default(cuid())
  personId      String
  orgName       String   // 'Nepal Education Foundation'
  title         String   // 'Program Coordinator'
  sector        String   // 'Education'
  type          String   // 'full_time_employment'|'freelance'
  start         DateTime
  end           DateTime?
  person        Person   @relation(fields: [personId], references: [id])
  experienceSkills ExperienceSkill[]
}

ExperienceSkill {
  experienceId  String
  skillId       String
  experience    Experience @relation(fields: [experienceId], references: [id])
  skill         Skill      @relation(fields: [skillId], references: [id])
  @@id([experienceId, skillId])
}

Education {
  id            String   @id @default(cuid())
  personId      String
  institution   String   // 'Tribhuvan University'
  university    String?
  level         String   // 'Bachelors'|'Masters'
  name          String   // 'Bachelor of Education'
  sector        String
  start         DateTime
  end           DateTime?
  person        Person   @relation(fields: [personId], references: [id])
}

Cohort {
  id            String   @id @default(cuid())
  name          String   // 'Tesro Paaila'
  fellowNStart  Int      // 60
  fellowNEnd    Int      // 58 (attrition)
  trainingN     Int      // 10 trainers
  start         DateTime
  fellowships   Fellowship[]
}

Fellowship {
  id            String   @id @default(cuid())
  cohortId      String
  personId      String
  years         Int      // 2
  start         DateTime
  end           DateTime?
  cohort        Cohort   @relation(fields: [cohortId], references: [id])
  person        Person   @relation(fields: [personId], references: [id])
  ldcAssignments LDC[]
  placedIn      PlacedIn[]
}

LDC {
  id            String   @id @default(cuid())
  personId      String   // LDC person
  fellowshipId  String
  person        Person   @relation("LDC_person", fields: [personId], references: [id])
  fellowship    Fellowship @relation(fields: [fellowshipId], references: [id])
}

Placement {
  id           String   @id @default(cuid())
  personId     String
  schoolId     String
  person       Person   @relation(fields: [personId], references: [id])
  school       School   @relation(fields: [schoolId], references: [id])
  placedIn     PlacedIn[]
}

School {
  id          String   @id @default(cuid())
  name        String   // 'Shree ABC Secondary School'
  address     String
  principal   String
  localId     String
  groupId     String
  localGov    LocalGov @relation(fields: [localId], references: [id])
  group       Group    @relation(fields: [groupId], references: [id])
  placements  Placement[]
}

LocalGov {
  id       String   @id @default(cuid())
  name     String   // 'Dang Municipality'
  province String   // 'Lumbini Province'
  schools  School[]
}

Group {
  id      String   @id @default(cuid())
  name    String   // 'Group 1 - Dang East'
  schools School[]
}

PlacedIn {
  id           String   @id @default(cuid())
  fellowshipId String
  placementId  String
  fellowship   Fellowship @relation(fields: [fellowshipId], references: [id])
  placement    Placement  @relation(fields: [placementId], references: [id])
}

Post {
  id          String   @id @default(cuid())
  personId    String
  title       String
  description String
  type        String   // 'hire'|'apply'
  createdAt   DateTime @default(now())
  person      Person   @relation(fields: [personId], references: [id])
}
